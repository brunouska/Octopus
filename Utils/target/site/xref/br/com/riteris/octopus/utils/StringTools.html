43       *
44       * @param stringToTest String a qual desejamos testar.

68      /**
69       * Remove os espaços em branco à esquerda de uma determinada String. Caso a String seja nula ou esteja vazia, a
70       * própria String será retornada sem nenhuma
71       * alteração.
72       *
73       * @param stringToTrim String a qual desejamos remover os espaços em branco à esquerda.
74       *
75       * @return String com todos os espaços em branco à esquerda removidos.
76       *
77       * @since 1.0.0 - Criada em 13 de out de 2015
78       */
79      public static String trimLeft( String stringToTrim ) {
80          if ( !StringTools.isNullOrEmpty( stringToTrim ) ) {
81              if ( stringToTrim.startsWith( " " ) ) {
82                  while ( stringToTrim.startsWith( " " ) ) {
83                      stringToTrim = stringToTrim.substring( 1 );
84                  }
85  
86                  return stringToTrim;
87              }
88              else {
89                  return stringToTrim;
90              }
91          }
92          else {
93              return stringToTrim;
94          }
95      }

97      /**
98       * Remove os espaços em branco à direita de uma determinada String. Caso a String seja nula ou esteja vazia, a
99  	 * própria String será retornada sem nenhuma
100      * alteração.
101      *
102      * @param stringToTrim String a qual desejamos remover os espaços em branco à direita.
103      *
104      * @return String com todos os espaços em branco à direita removidos.
105      *
106      * @since 1.0.0 - Criada em 13 de out de 2015
107      */
108     public static String trimRight( String stringToTrim ) {
109         if ( !StringTools.isNullOrEmpty( stringToTrim ) ) {
110             if ( stringToTrim.endsWith( " " ) ) {
111                 while ( stringToTrim.endsWith( " " ) ) {
112                     stringToTrim = stringToTrim.substring( 0, stringToTrim.length() - 1 );
113                 }
114 
115                 return stringToTrim;
116             }
117             else {
118                 return stringToTrim;
119             }
120         }
121         else {
122             return stringToTrim;
123         }
124     }

128 	 * esteja vazia, a própria String será retornada

130      *
131      * @param stringToTrim String a qual desejamos remover os espaços em branco à esquerda e à direita.

133      * @return String com todos os espaços em branco à esquerda e à direita removidos.
134      *
135      * @since 1.0.0 - Criada em 13 de out de 2015

137     public static String trim( String stringToTrim ) {
138         if ( !StringTools.isNullOrEmpty( stringToTrim ) ) {
139             stringToTrim = StringTools.trimLeft( stringToTrim );
140             stringToTrim = StringTools.trimRight( stringToTrim );
141 
142             return stringToTrim;
143         }
144         else {
145             return stringToTrim;
146         }

149     /**
150      * Acrescenta espaços em branco à esquerda de uma determinada String até que seja atingido o tamanho desejado.
151 	 * Caso a String seja maior do que o tamanho
152      * desejado ou o tamanho desejado for 0 ou negativo, a própria String será retornada sem nenhuma alteração. Caso a
153      * String seja nula, será criada uma String
154      * vazia e preenchida com os espaços em branco desejados conforme os parâmetros informados ou retornada vazia,
155      * caso esteja nula e os parâmetros informados
156      * não permitem o seu preenchimento.
157      *
158      * @param stringToFillWithBlankSpaces String a qual serão acrescentados espaços em branco à sua esquerda até que o
159      *                                       tamanho desejado seja atingido.
160      * @param desiredLength               Tamanho desejado para a String resultante.
161      *
162      * @return String preenchida com espaços em branco à sua esquerda com o tamanho informado por parâmetro.
163      *
164      * @since 1.0.0 - Criada em 13 de out de 2015
165      */
166     public static String fillLeftWithBlankSpacesToDesiredLength( String stringToFillWithBlankSpaces, final int
167             desiredLength ) {
168         if ( StringTools.isNull( stringToFillWithBlankSpaces ) ) {
169             stringToFillWithBlankSpaces = "";
170         }
171 
172         if ( ( desiredLength <= 0 ) || ( stringToFillWithBlankSpaces.length() >= desiredLength ) ) {
173             return stringToFillWithBlankSpaces;
174         }
175         else {
176             int blankSpacesNumberToAdd = desiredLength - stringToFillWithBlankSpaces.length();
177 
178             return StringTools.addBlankSpaces( stringToFillWithBlankSpaces, blankSpacesNumberToAdd, true, false );
179         }
180     }

195     /**
196      * Adiciona espaços em branco no início ou ao final de uma determinada String, baseado no número de espaços a
197 	 * serem inseridos e baseado na configuração de
198      * posição (esquerda e/ou direita). Caso o numero de espaços seja 0 ou negativo ou não tenha sido escolhida
199      * nenhuma das opções de preenchimento (esquerda ou
200      * direita), a própria String será retornada sem nenhuma alteração. Caso a String seja nula, será criada uma
201      * String vazia e preenchida com os espaços em
202      * branco desejados conforme os parâmetros informados.
203      *
204      * @param stringToAddBlankSpaces String a qual serão adicionados os espaços em branco.
205      * @param numberOfSpaces         Número de espaços em branco os quais serão adicionados.
206      * @param atLeft                 Indica que devem ser adicionados espaços em branco à esquerda do início da String.
207      * @param atRight                Indica que devem ser adicionados espaços em branco à direita do término da String.
208      *
209      * @return String com os espaços em branco adicionados conforme as configurações informadas.
210      *
211      * @since 1.0.0 - Criada em 13 de out de 2015
212      */
213     public static String addBlankSpaces( String stringToAddBlankSpaces, final int numberOfSpaces, final boolean
214             atLeft, final boolean atRight ) {
215         if ( numberOfSpaces <= 0 || ( !atLeft && !atRight ) ) {
216             return stringToAddBlankSpaces;
217         }
218         else {
219             if ( StringTools.isNull( stringToAddBlankSpaces ) ) {
220                 stringToAddBlankSpaces = "";
221             }
222 
223             if ( atLeft ) {
224                 for ( int i = 0; i < numberOfSpaces; i++ ) {
225                     stringToAddBlankSpaces = " " + stringToAddBlankSpaces;
226                 }
227             }
228 
229             if ( atRight ) {
230                 for ( int i = 0; i < numberOfSpaces; i++ ) {
231                     stringToAddBlankSpaces += " ";
232                 }
233             }
234 
235             return stringToAddBlankSpaces;
236         }
237     }

272     /**
273      * Extrai de uma determinada String os caracteres que estiverem delimitados pelos delimitadores informados, não
274 	 * incluindo os mesmos. Caso a String informada
275      * seja nula, vazia ou esteja em branco ou seus delimitadores, será retornada uma String vazia. Caso os
276      * delimitadores apontem para a mesma posição ou o
277      * delimitador final ocorra antes do delimitador inicial na String informada, será retornada uma String vazia. O
278      * trecho de caracteres que serão extraídos da
279      * String serão somente os caracteres correspondentes ao primeiro conjunto de delimitadores encontrados.
280      *
281      * @param sourceString   String da qual desejamos extrair os caracteres delimitados pelos delimitadores informados.
282      * @param firstDelimiter Delimitador inicial do trecho o qual desejamos extrair.
283      * @param lastDelimiter  Delimitador final do trecho o qual desejamos extrair.
284      *
285      * @return Caracteres extraídos da String informada, delimitados pelos delimitadores informados.
286      *
287      * @since 1.0.0 - Criada em 13 de out de 2015
288      */
289     public static String extractPart( final String sourceString, final String firstDelimiter, final String
290             lastDelimiter ) {
291         if ( StringTools.isNullEmptyOrBlank( sourceString ) ) {
292             return "";
293         }
294         else if ( StringTools.isNullEmptyOrBlank( firstDelimiter ) || StringTools.isNullEmptyOrBlank( lastDelimiter )
295 				) {
296             return "";
297         }
298         else {
299             int beginIndex = ( sourceString.indexOf( firstDelimiter ) + firstDelimiter.length() );
300             int endIndex = sourceString.indexOf( lastDelimiter );
301 
302             if ( ( beginIndex < 0 || endIndex < 0 ) || ( beginIndex >= endIndex ) ) {
303                 return "";
304             }
305             else {
306                 return sourceString.substring( beginIndex, endIndex );
307             }
308         }
309     }

311     /**
312      * Extrai de uma determinada String os caracteres que estiverem delimitados pelos delimitadores informados, não
313 	 * incluindo os mesmos. Caso a String informada
314      * seja nula, vazia ou esteja em branco ou seus delimitadores, será retornada uma listagem vazia. Serão extraídos
315      * todos os caracteres contidos nos conjuntos
316      * de delimitadores encontrados na String, seguindo as mesmas regras do método
317      * {@link #extractPart(String, String, String) extractPart} para cada conjunto
318      * de delimitadores encontrados, os quais serão colocados na lista de caracteres que será retornada.
319      *
320      * @param sourceString   String da qual desejamos extrair os caracteres delimitados pelos delimitadores informados.
321      * @param firstDelimiter Delimitador inicial do trecho o qual desejamos extrair.
322      * @param lastDelimiter  Delimitador final do trecho o qual desejamos extrair.
323      *
324      * @return Listagem contendo os caracteres extraídos da String informada, delimitados pelos delimitadores
325      * informados.
326      *
327      * @since 1.0.0 - Criada em 13 de out de 2015
328      */
329     public static Collection< String > extractParts( String sourceString, final String firstDelimiter, final String
330             lastDelimiter ) {
331         Collection< String > result = new ArrayList<>();
332 
333         if ( StringTools.isNullEmptyOrBlank( sourceString ) ) {
334             return result;
335         }
336         else if ( StringTools.isNullEmptyOrBlank( firstDelimiter ) || StringTools.isNullEmptyOrBlank( lastDelimiter )
337 				) {
338             return result;
339         }
340         else {
341             while ( sourceString.contains( firstDelimiter ) && sourceString.contains( lastDelimiter ) ) {
342                 result.add( StringTools.extractPart( sourceString, firstDelimiter, lastDelimiter ) );
343                 sourceString = sourceString.replaceFirst( firstDelimiter, "" );
344                 sourceString = sourceString.replaceFirst( lastDelimiter, "" );
345             }
346 
347             return result;
348         }
349     }

351     /**
352      * Dada uma determinada String, troca todos os trechos localizados entre os conjuntos de delimitadores pelos
353 	 * valores correspondentes existentes no mapa
354      * informado, sendo os valores encontrados as chaves dos valores dos mapas. Caso a String informada esteja nula,
355 	 * vazia ou em branco ou o mapa e os
356      * delimitadores, será retornada a String informada sem qualquer tipo de alteração.
357      *
358      * @param sourceString   String para a qual desejamos substituir os trechos localizados entre os conjuntos de
359      *                          delimitadores.
360      * @param targetPartsMap Mapa contendo os valores para substituição na String informada com as chaves localizadas
361      *                          entre os delimitadores.
362      * @param firstDelimiter Delimitador do início do trecho a ser substituído.
363      * @param lastDelimiter  Delimitador do final do trecho a ser substituído.
364      *
365      * @return String com os trechos localizados entre os conjuntos de delimitadores pelos valores contidos no mapa
366      * informado.
367      *
368      * @since 1.0.0 - Criada em 14 de out de 2015
369      */
370     public static String replacePartsWithMap( String sourceString, final Map< String, String > targetPartsMap, final
371     String firstDelimiter,
372                                               final String lastDelimiter ) {
373         if ( StringTools.isNullEmptyOrBlank( sourceString ) ) {
374             return sourceString;
375         }
376         else if ( CollectionAndMapTools.mapIsNullOrEmpty( targetPartsMap ) ) {
377             return sourceString;
378         }
379         else if ( StringTools.isNullEmptyOrBlank( firstDelimiter ) || StringTools.isNullEmptyOrBlank( lastDelimiter ) ) {
380             return sourceString;
381         }
382         else {
383             final Collection< String > extractedParts = StringTools.extractParts( sourceString, firstDelimiter, lastDelimiter );
384 
385             if ( !CollectionAndMapTools.collectionIsNullOrEmpty( extractedParts ) ) {
386                 for ( String extractedPart : extractedParts ) {
387                     if ( targetPartsMap.containsKey( extractedPart ) ) {
388                         sourceString = sourceString.replace( ( firstDelimiter + extractedPart + lastDelimiter ), targetPartsMap.get( extractedPart ) );
389                     }
390                 }
391             }
392 
393             return sourceString;
394         }
