6   /**
7    * Conjunto de métodos utilitários para a manipulação de coleções e mapas. Este objeto, por tratar-se de um conjunto
8    * de utilitários, foi projetado para se
9    * integrar de forma transparente ao código fonte fornecendo métodos utilitários, os quais foram projetados com
10   * programação defensiva para não lançarem exceções
11   * em seu funcionamento interno.
12   *
13   * @author Bruno Barauskas
14   * @version 1.0.0 - Criação do objeto.
15   * @since 1.0.0 - Criada em 14 de out de 2015
16   */
17  public final class CollectionAndMapTools {
18  
19      /**
20       * Construtor do objeto, com visibilidade privada, pois o objeto é um utilitário portador de métodos estáticos.
21       *
22       * @since 1.0.0 - Criada em 14 de out de 2015
23       */
24      private CollectionAndMapTools() {
25      }
26  
27      /**
28       * Testa qualquer objeto do tipo java.util.Collection para verificar se a coleção está nula ou vazia.
29       *
30       * @param collectionToEval Coleção para o teste.
31       *
32       * @return True caso a coleção testada esteja nula ou vazia e false caso contrário.
33       *
34       * @since 1.0.0 - Criada em 14 de out de 2015
35       */
36      public static boolean collectionIsNullOrEmpty( Collection< ? > collectionToEval ) {
37          return collectionToEval == null || collectionToEval.isEmpty();
38      }
39  
40      /**
41       * Testa qualquer objeto do tipo java.util.Map para verificar se o mapa está nulo ou vazio.
42       *
43       * @param mapToEval Mapa para o teste.
44       *
45       * @return True caso o mapa testado esteja nulo ou vazio e false caso contrário.
46       *
47       * @since 1.0.0 - Criada em 14 de out de 2015
48       */
49      public static boolean mapIsNullOrEmpty( Map< ?, ? > mapToEval ) {
50          return mapToEval == null || mapToEval.isEmpty();
51      }
52  
53  }
