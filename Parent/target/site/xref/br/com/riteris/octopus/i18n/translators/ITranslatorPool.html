5   /**
6    * Define o comportamento do objeto responsável por fornecer tradutores configurados com os requisitos necessários ao
7    * seu funcionamento e prover métodos para o
8    * gerenciamento dos recursos de internacionalização.
9    *
10   * @author Bruno Barauskas
11   * @version 1.0.0 - Criação da interface.
12   * @since 1.0.0 - Criada em 17 de abr de 2016
13   */
14  public interface ITranslatorPool {
15  
16      /**
17       * Método utilizado para fazr a atualização dos recursos de internacionalização para uma determinada linguagem. A
18       * atualização se dá pelo processo de
19       * recarregamento dos recursos por parte do pool.
20       *
21       * @param language Linguagem para a qual faremos a atualização dos recursos de internacionalização disponíveis.
22       *
23       * @since 1.0.0 - Criada em 18 de abr de 2016
24       */
25      void reloadResouceBundle( Language language );
26  
27      /**
28       * Fornece um tradutor para a linguagem desejada, informada como parâmetro. Caso não seja localizado um tradutor
29       * para a linguagem informada, será fornecido
30       * um tradutor para a linguagem padrão, ou nulo caso nenhum tradutor possa ser criado.
31       *
32       * @param language Linguagem para a qual desejamos um tradutor devidamente configurado.
33       *
34       * @return Tradutor configurado para a linguagem informada como parâmetro, pronto para utilização ou nulo caso
35       * nenhum tradutor possa ser criado.
36       *
37       * @since 1.0.0 - Criada em 18 de abr de 2016
38       */
39      ITranslator getReadyTranslator( Language language );
40  
41  }
