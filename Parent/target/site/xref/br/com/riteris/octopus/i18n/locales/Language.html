165     /**
166      * Dado um código completo de idioma e país, retorna o valor do enum {@link Language} correspondente ao código
167      * informado, ou nulo caso não seja localizado o
168      * idioma correspondente.
169      *
170      * @param completeLanguageCode Código completo do idioma o qual desejamos obter o valor do enum {@link Language}
171      *                                correspondente.
172      *
173      * @return Valor do enum {@link Language} correspondente ao código de idioma informado ou nulo caso não seja
174      * localizado um idioma correspondente ao código.
175      *
176      * @since 1.0.0 - Criada em 18 de abr de 2016
177      */
178     public final static Language getLanguageByCode( String completeLanguageCode ) {
179         if ( StringTools.isNullEmptyOrBlank( completeLanguageCode ) ) {
180             throw new IllegalArgumentException( "The complete language code must be informed." );
181         }
182 
183         completeLanguageCode = completeLanguageCode.replace( "-", "_" );
184         completeLanguageCode = completeLanguageCode.replace( "/", "_" );
185 
186         for ( Language language : Language.values() ) {
187             if ( language.name().toUpperCase().equals( completeLanguageCode.toUpperCase() ) ) {
188                 return language;
189             }
190         }
191 
192         return null;
193     }
