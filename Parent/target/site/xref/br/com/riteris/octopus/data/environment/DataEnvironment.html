1   package br.com.riteris.octopus.data.environment;
2   
3   import br.com.riteris.octopus.utils.StringTools;
4   
5   /**
6    * Enum contendo os valores possíveis para os ambientes de dados dos repositórios.
7    *
8    * @author Bruno Barauskas
9    * @version 1.0.0 - Criação do ENUM.
10   * @since 1.0.0 - Criada em 6 de mar de 2016
11   */
12  public enum DataEnvironment {
13  
14      PD( "Production", "ROLE_ACCESS_PD" ),
15      HL( "Homologation", "ROLE_ACCESS_HL" ),
16      DV( "Development", "ROLE_ACCESS_DV" );
17  
18      /**
19       * Descrição do ambiente de dados.
20       */
21      private final String description;
22  
23      /**
24       * Descrição da regra de acesso necessária para utilizar o ambiente de dados.
25       */
26      private final String accessRole;
27  
28      /**
29       * Construtor do objeto.
30       *
31       * @param description {@link #description}.
32       * @param accessRole  {@link #accessRole}.
33       *
34       * @since 1.0.0 - Criada em 6 de mar de 2016
35       */
36      DataEnvironment( final String description, final String accessRole ) {
37          this.description = description;
38          this.accessRole = accessRole;
39      }
40  
41      /**
42       * Fornece uma descrição detalhada do ambiente de dados, contendo a sua sigla e a sua respectva descrição.
43       *
44       * @return Descrição detalhada do ambiente de dados e a sua respectiva descrição.
45       *
46       * @since 1.0.0 - Criada em 6 de mar de 2016
47       */
48      public final String getCompleteName() {
49          return ( this.name() + "-" + this.description );
50      }
51  
52      /**
53       * Obtém o ambiente de dados correspondente à regra de acesso informada. Caso a regra de acesso não seja informada
54       * corretamente ou não existe nenhum
55       * ambiente de dados correspondente à regra de acesso será retornado nulo.
56       *
57       * @param accessRole Regra de acesso à qual desejamos obter o ambiente de dados correspondente.
58       *
59       * @return Ambiente de dados correspondente à regra de acesso informada ou nulo caso a regra de acesso não seja
60       * informada corretamente ou não exista um
61       * ambiente de dados correspondente.
62       *
63       * @since 1.0.0 - Criada em 6 de mar de 2016
64       */
65      public final DataEnvironment getEnvironmentByAccessRole( final String accessRole ) {
66          if ( StringTools.isNullEmptyOrBlank( accessRole ) ) {
67              return null;
68          }
69  
70          for ( DataEnvironment dataEnvironment : DataEnvironment.values() ) {
71              if ( dataEnvironment.accessRole.equals( accessRole ) ) {
72                  return dataEnvironment;
73              }
74          }
75  
76          return null;
77      }
78  
79  }
