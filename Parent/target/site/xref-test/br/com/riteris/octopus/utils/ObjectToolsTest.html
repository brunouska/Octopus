102      * Test method for
103      * {@link br.com.riteris.octopus.utils.ObjectTools#areEqualsUsingProperty(java.lang.Object, java.lang.Object, java.util.Collection)}
104      * .
105      */
106     @Test
107     public final void testAreEqualsUsingProperty() {
108         assertFalse( ObjectTools.areEqualsUsingProperty( stringNula, stringNula, propsWithOneProp ) );
109         assertFalse( ObjectTools.areEqualsUsingProperty( stringNula, stringVazia, propsWithOneProp ) );
110         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringNula, propsWithOneProp ) );
111         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringVazia, propsEmpty ) );
112         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringVazia, propsNull ) );
113         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringEmBranco, propsWithOneProp ) );
114         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringPreenchida, propsWithOneProp ) );
115         assertFalse( ObjectTools.areEqualsUsingProperty( stringVazia, stringEmBranco, propsWithOneProp ) );
116         assertFalse( ObjectTools.areEqualsUsingProperty( stringEmBranco, stringPreenchida, propsWithOneProp ) );
117         assertFalse( ObjectTools.areEqualsUsingProperty( stringPreenchida, stringComEspacoEmBrancoEsq,
118                 propsWithOneProp ) );
119         assertFalse( ObjectTools.areEqualsUsingProperty( stringPreenchida, stringComEspacoEmBrancoEsq,
120                 propsWithTwoProp ) );
121         assertTrue( ObjectTools.areEqualsUsingProperty( stringPreenchida, stringPreenchida, propsWithOneProp ) );
122         assertTrue( ObjectTools.areEqualsUsingProperty( stringPreenchida, stringPreenchida, propsWithTwoProp ) );
123     }

128     @Test
129     public final void testGetObjectDescriptionResume() {
130         assertEquals( ObjectTools.getObjectDescriptionResume( null ), "N/A: -NULL-" );
131         assertEquals( ObjectTools.getObjectDescriptionResume( stringNula ), "N/A: -NULL-" );
132         assertEquals( ObjectTools.getObjectDescriptionResume( stringVazia ), "String: -EMPTY-" );
133         assertEquals( ObjectTools.getObjectDescriptionResume( stringEmBranco ), "String: -BLANK-" );
134         assertEquals( ObjectTools.getObjectDescriptionResume( stringPreenchida ), "String: Teste" );
135         assertEquals( ObjectTools.getObjectDescriptionResume( collectionPreenchida ), "ArrayList: [ String: A, " +
136                 "String: B, String: C ]" );
137         assertEquals( ObjectTools.getObjectDescriptionResume( collectionVazia ), "ArrayList: [ -EMPTY- ]" );
138         assertEquals( ObjectTools.getObjectDescriptionResume( collectionNula ), "N/A: -NULL-" );
139         assertEquals( ObjectTools.getObjectDescriptionResume( mapaPreenchida ), "HashMap: [ Key: String: 1 - Value: " +
140                 "String: A, Key: String: 2 - Value: String: B, Key: String: 3 - Value: String: C ]" );
141         assertEquals( ObjectTools.getObjectDescriptionResume( mapaVazio ), "HashMap: [ -EMPTY- ]" );
142         assertEquals( ObjectTools.getObjectDescriptionResume( mapaNulo ), "N/A: -NULL-" );
143         assertEquals( ObjectTools.getObjectDescriptionResume( mapaComplexoUmNivel ), "HashMap: [ Key: String: 1A - " +
144                 "Value: ArrayList: [ String: AA, String: BB, String: CC ], Key: String: 1B - Value: ArrayList: [ " +
145                 "String: AAA, String: BBB, String: CCC ], Key: String: 1C - Value: ArrayList: [ String: AAAA, String: BBBB, String: CCCC ] ]" );
146         assertEquals( ObjectTools.getObjectDescriptionResume( mapaComplexoDoisNiveis ), "HashMap: [ Key: String: 2B - Value: HashMap: [ Key: String: 1AA - Value: ArrayList: [ String: AA, String: BB, String: CC ] ], Key: String: 2A - Value: HashMap: [ Key: String: 1A - Value: ArrayList: [ String: AA, String: BB, String: CC ] ] ]" );
147     }
